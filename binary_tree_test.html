Создадим класс Node и объявим корневой узел root
Корень дерева (а в данном случае списка) инициализируется числовым значением с помощью конструктора класса Node, принимающим в качестве параметра переменную int value
Метод display используется для отображения вершины 

   Node root;
   void setup(){
     background(100);
     size(300, 300);
       root = new Node(5);
       root.display();
   }
   class Node{
   int value;
   int y = 100;
   //constructor
   Node(int value){
     this.value = value;
     }
   
   void display(){
     textSize(40);
     text(value, 145, y);
     }
   }
Метод add принимает в качестве параметра новый экземпляр класса, то есть элемент типа Node, созданный конструктором класса, сдвигает его вниз по оси y, затем заменяет корневой элемент новым узловым элементом, полученным в качестве параметра 
  Node root;
   void setup(){
     background(100);
     size(300,300);
       root = new Node(5);
       root.display();
           root.add( new Node(3) );
	   root.display();
	   root.add( new Node(7) );
	   root.display();
		 
   }
   class Node{
   int value;
   int y = 100;
   //constructor
   Node(int value){
     this.value = value;
     }
   void add(Node node){
     node.y = y + 40;
     root = node;
   }  
   void display(){
     textSize(40);
     text(value, 145,y);
     }
   }
Визуализация алгоритма отрисовки списка из трёх элементов

Вообще, метод add может принимать переменную int, которая затем обрабатывается конструктором класса внутри вспомогательного метода Node convert()
Переменная node используется методом convert для создания нового экземпляра класса 
Node root;
   void setup(){
     background(100);
     size(300,300);
       root = new Node(5);
       root.display();      
       root.add(3);
       root.display();     
       root.add(7);
       root.display();
     
   }
   class Node{
   int value;
   int y = 50;
   Node node = null;
   //constructor
   Node(int value){
     this.value = value;
     }
   
   Node convert(Node node, int value){
   if(node == null) node = new Node(value);
   return node;
   }
   
    void add(int value){ 
     root = convert(node, value);
     root.y = this.y + 40;
   }  
   void display(){
     textSize(40);
     text(value, 145,y);
     }
   }
