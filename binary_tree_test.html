Создадим класс Node и объявим корневой узел root 

   Node root;
   void setup(){
     background(100);
     size(150,150);
       root = new Node(5);
       root.display();
   }
   class Node{
   int value;
   //constructor
   Node(int value){
     this.value = value;
     }
   
   void display(){
     textSize(40);
     text(value, 65, 80);
     }
   }
